18  public class MowerParser implements Parser<String, Mower> {
19  
20  	private static final String SEPARATOR = " ";
21  	private static final String PATTERN = "^\\d+ \\d+ [N|E|W|S]$";
22  
23  	private final Lawn lawn;
24  	
25  	private final AtomicInteger counter = new AtomicInteger();
26  	
27  	public MowerParser(Lawn lawn) {
28  		this.lawn = lawn;
29  	}
30  	
31  	@Override
32  	public Mower parse(String source) {
33  		checkArgument(source.matches(PATTERN),
34  		        "Error when parsing the mower infos [Expecting: Positive integer 'X Y' and 'Orientaion'; Got: '"
35  		            + source + "']");
36  		    List<String> fields = Splitter.on(SEPARATOR).splitToList(source);
37  		    int x = Integer.valueOf(fields.get(0));
38  		    int y = Integer.valueOf(fields.get(1));
39  		    Orientation orientation = Orientation.valueOf(fields.get(2));
40  		    return new Mower(counter.getAndIncrement(), lawn.cellAt(x, y), orientation);
41  	}
